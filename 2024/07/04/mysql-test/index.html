创建触发器的代码：
create trigger deleteemployees after delete on departments
for each row
begin
delete from employees where departmentid=old.departmentid;
delete from salary where employeeid in(select employeeid from employees where departmentid=old.departmentid);
end;

游标：创建一个存储过程，输入课程号，求选修该课程的学生人数和学生姓名
create procedure getnumandname(in cno char(10),out num int,out xsnames varchar(500))
begin
declare flag boolean default true;#定义游标循环标志
declare sno char(10);#定义游标查询语句中选项的变量
declare sname char(10);#定义游标查询语句中选项的变量
declare num_c cursor for#定义游标
select sc.学号,姓名 from sc.student where sc.学号=student.学号 and 课程号=cno;
declare exit handler for not found set flag=false;#如果游标没有下一行数据，flag=false不循环
set num=0; #必须线set必须卸载所以declare语句后面
set xsnames=’’; #必须线set必须卸载所有declare语句后面
open num_c; #打开游标
while flag do #循环打开游标
fetch num_c into sno,sname; #游标按行抓取值
set num=num+1; #人数加1
set xsnames=concat(xsnames,’ ‘,sname); #人名累计添加
end while;
close num_c; #关闭游标
end;

call getnumandname(‘102’,@num,@xsname);
select @num,@xsname;
